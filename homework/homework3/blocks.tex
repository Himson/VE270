\newcommand{\drawaddermodel}[3]{
	\draw #1 node (#2) [shape=rectangle,draw,minimum height=2cm,minimum width=4cm,text width=3cm,align=center] {#3};
	\draw (#2) ++(left:17.5mm) node {$c_o$};
	\draw (#2) ++(right:17.5mm) node {$c_i$};
	\draw (#2) ++(down:7.5mm) node {$s$};
	\draw (#2) ++(up:7.5mm) ++(left:10mm) node {$a$} ++(right:20mm) node {$b$};
	\draw (#2) ++(down:10mm) to node {$\diagup$} node[left] {8} ++(down:5mm) ;
}

\newcommand{\drawadder}[4]{
	\drawaddermodel{#1}{#2}{#3}
	\draw (#2) ++(up:10mm) ++(left:10mm) to node {$\diagup$} node[left] {8} ++(up:5mm) ;
	\draw (#2) ++(up:10mm) ++(right:10mm) to node {$\diagup$} node[left] {8} ++(up:10mm) ;
	\draw (#2.east) ++(right:10mm) node (#2__ci) {0};
	\draw (#2.east) -- (#2__ci);
}

\newcommand{\drawsubtractor}[4]{

	\draw (#2) ++(up:10mm) ++(left:10mm) to node {$\diagup$} node[left] {8} ++(up:5mm);
	\draw (#2) ++(up:15mm) ++(right:20mm) node[not gate,point left] (#2__notgate) {\tiny #4bit};
	\draw (#2) ++(up:10mm) ++(right:10mm) to node {$\diagup$} node[left] {#4} ++(up:5mm) -- (#2__notgate.output);
	\draw (#2__notgate.input) -- ++(right:5mm) to node {$\diagup$} node[left] {#4} ++(up:5mm);
	\draw (#2.east) ++(right:10mm) node (#2__ci) {1};
	\draw (#2.east) -- (#2__ci);
}
